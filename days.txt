# Настройки анализа временного периода
# Параметр offset_days задает смещение дней от текущей даты для определения начальной точки анализа.
# Например, offset_days = 7 будет указывать на начало недели перед текущей датой (текущая неделя минус семь дней).
offset_days=7

# Параметр days указывает продолжительность периода анализа, в днях.
# Например, days = 6 задает длительность периода в 6 дней с учетом offset_days, захватывая всю прошлую неделю.
days=6
